## Installation
```
npm install @comento-ui

yarn add @comento-ui
```

### How to use
```vue
import Vue from 'vue';
import ComentoUI from '@comento-ui';
import '@comento/ui/dist/@comento-ui.css';

Vue.use(ComentoUI);
```
