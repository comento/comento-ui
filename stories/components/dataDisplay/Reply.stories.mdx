import { ArgsTable, Meta, Story, Canvas } from '@storybook/addon-docs/blocks';
import Reply from '@/components/components/dataDisplay/Reply';
import UserInformation from '@/components/components/dataDisplay/content/UserInformation';
import Avatar from '@/components/components/dataDisplay/Avatar';
import Icon from '@/components/elements/core/icon/Icon';
import { colorKeys } from "@/utils/constants/color";

<Meta
	title="Data Display/Reply"
	component={Reply}
	argTypes={{
		value: {
			description: '댓글 내용',
			defaultValue: '댓글 내용이 표시됩니다.',
			control: { type: 'text' },
		},
		fontColor: {
			description: '댓글 색상',
      control: {
        type: 'select',
        options: colorKeys,
      },
		},
		date: {
			description: '날짜',
			defaultValue: '21.09.02',
			control: { type: 'text' },
		},
		type: {
			description: '댓글 타입',
			defaultValue: 'default',
			control: {
				type: 'select',
				options: ['default', 'simple'],
			},
		},
		showReport: {
			description: '신고 버튼 보여주기',
			control: {
				type: 'boolean',
			},
		},
	}}
/>

export const Basic = (args, { argTypes }) => ({
	props: Object.keys(argTypes),
	components: { Reply, UserInformation, Avatar, Icon },
	data() {
		return {
			name: '닉네임',
			company: '코멘토',
		};
	},
	computed: {
		avatarSize() {
			const sizes = {
				default: 'small',
				simple: 'xsmall',
			};
			return sizes[this.type];
		},
		userInformationType() {
			if (this.type === 'default') {
				return 'normal';
			}
			return this.type;
		},
	},
	methods: {
		openReportModal() {
			alert('신고하기 버튼 클릭');
		},
	},
	template: `
        <Reply :value="value" :date="date" :font-color="fontColor" :show-report="showReport" @clickReportButton="openReportModal">
            <template v-slot:user-information>
                <UserInformation :type="userInformationType">
                    <template v-slot:avatar>
                        <Avatar type="nickname" :text="name.substr(0, 1)" :size="avatarSize"/>
                    </template>
                    <template v-slot:name>
                        {{ name }}
                    </template>
                    <template v-slot:icon>
                        <Icon name="IconCheckRoundSmallFill" color="primary"/>
                    </template>
                    <template v-slot:company>
                        {{ company }}
                    </template>
                </UserInformation>
            </template>
        </Reply>`,
});
