export const Basic = (args, { argTypes }) => ({
    props: Object.keys(argTypes),
    components: { CUserInformation, CContent, CAvatar, CIcon, CChipGroup, CChip },
    data() {
        return {
            matchedItems: ['회사', '산업', '직무', '학교'],
        };
    },
    computed: {
        avatarSize() {
            const sizes = {
                simple: 'xsmall',
                normal: 'small',
                full: 'medium'
            };
            return sizes[this.type];
        }
    },
    template: `
        <CUserInformation :type="type">
            <template v-slot:avatar>
                <CAvatar type="text" :text="name.substr(0, 1)" :size="avatarSize"/>
            </template>
            <template v-slot:name>
                {{ name }}
            </template>
            <template v-slot:icon>
                <CIcon name="IconCheckSmallFill" color="primary"/>
            </template>
            <template v-slot:company>
                {{ company }}
            </template>

                코부사장 ∙ 채택률 90% ∙

                        v-for="(item, index) in matchedItems"
                        :key="'chip' + index"
                        type="outlined"
                        color="primary"
                        size="small"
                    >
                        {{ item }}
                    </CChip>
                </CChipGroup>
                일치
            </template>
        </CUserInformation>`,
});
