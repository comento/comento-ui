export const Basic = (args, { argTypes }) => ({
	props: Object.keys(argTypes),
	components: {
		Reply,
		UserInformation,
		UserInformationName,
		UserInformationCompany,
		UserInformationCaption,
		Avatar,
		Typography,
		Icon,
		ReplyActionsExample,
		ReplyInteractionsExample,
	},
	data() {
		return {
			name: '닉네임',
			company: '코멘토',
		};
	},
	computed: {
		avatarSize() {
			const sizes = {
				default: 'small',
				simple: 'xsmall',
			};
			return sizes[this.type];
		},
		userInformationType() {
			const types = {
				default: 'normal',
				simple: 'simple',
			};
			return types[this.type];
		},
	},
	methods: {
		nl2br,
	},
	template: `
        <Reply>
            <template v-slot:user-information>
				<UserInformation :type="userInformationType">
					<template v-slot:avatar>
						<Avatar type="nickname" :text="name.substr(0, 1)" :size="avatarSize" />
					</template>
					<template v-slot:name>
						<UserInformationName :type="userInformationType">
							{{ name }}
						</UserInformationName>
					</template>
					<template v-slot:user-info>
						<UserInformationCompany :type="userInformationType">
							{{ company }}
						</UserInformationCompany>
					</template>
					<template v-slot:other>
						<UserInformationCaption>2025.01.01</UserInformationCaption>
					</template>
				</UserInformation>
            </template>
			<template v-slot:value>
				<span v-html="nl2br('댓글 내용이 표시됩니다.<br />링크는 이렇게 표시 됩니다. https://comento.kr')" />
			</template>
			<template v-slot:actions>
				<ReplyActionsExample />
			</template>
			<template v-slot:interactions>
				<ReplyInteractionsExample/>
			</template>
        </Reply>`,
});
