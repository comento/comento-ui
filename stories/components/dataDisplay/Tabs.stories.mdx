<Meta

export const Default = (args, { argTypes }) => ({
    props: Object.keys(argTypes),
    components: { CTabs, CButton },
    data() {
        return {
            selectedNo: 2,
            items: ['캠프 일정', '새 댓글', '수료증'],
        }
    },
    methods: {
        handleChangeTabIndex(index) {
            this.selectedNo = index
        }
    },
    template:
      `<div style="background: cadetblue">
          <CTabs :type="type" :direction="direction" :tab-index="selectedNo" :transparent="transparent" @changeTabIndex="handleChangeTabIndex">
              <template v-for="(item, index) in items" :slot="'item' + index">
                    <CButton>
                      {{ item }}
                    </CButton>
              </template>
          </CTabs>
        </div>`,
});
