export const Default = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  data() {
      return {
          files: [],
      };
  },
  methods: {
    handleUploadFiles(files) {
        const filesWithLoading = Object.values(files).map(file => ({file: file, isLoading: false}));
        this.files = [...this.files, ...filesWithLoading];
    },
    handleClickFileDownloadIcon(payload) {
        this.files[payload.index].isLoading = true;
		setTimeout(() => {
            this.files[payload.index].isLoading = false;
        }, 3000);
    },
    handleClickFileItemContent(payload) {
        alert('clickItemContent')
    },
    handleClickFileTrashIcon(payload) {
        this.files = this.files.filter((item, index) => index !== payload.index);
    }
  },
  template: `<div style="width: 100%; display: flex; justify-content: center; align-items: flex-start; flex-direction: column;">
                 <FileInputButton
                    :accept="accept"
                    :multiple="multiple"
                    :full="full"
                    :button-text="buttonText"
                    :disabled="disabled"
                    @upload-files="handleUploadFiles"
                 />
                 <FileInputList
                    :full="full"
                    @download-item="handleClickFileDownloadIcon"
                    @click-item-content="handleClickFileItemContent"
                    @remove-item="handleClickFileTrashIcon"
                 >
					<FileInputFileItem
						v-for="(item, index) in files"
						:key="'file-' + index"
						:file="item.file"
						:index="index"
						is-removable
						:is-loading="item.isLoading"
						:is-error="item.isError"
						@remove-item="handleClickFileTrashIcon"
					/>
				</FileInputList>
              </div>`,
   components: { FileInputButton, FileInputList, FileInputFileItem },
});
