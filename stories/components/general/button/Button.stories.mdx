<Canvas>
    <Story name="All" height="100px">
        {{
            template: `
                <div>
                  <template v-for="(type, index) in buttonTypes">
                    <div class="flex">
                        <CButtonStateTitle class="mr-30">{{type}}</CButtonStateTitle>
                        <div class="flex-column mr-30">
                            <CButtonStateTitle>default</CButtonStateTitle>
                            <div :style="getGhostTypeBackgroundColor(type)">
                                <template v-for="(color, index) in getFilterButtonColors(type)">
                                    <CButton :color="color" :type="type" class="mb-16">Button</CButton>
                                </template>
                            </div>
                        </div>
                        <div class="flex-column mr-30">
                            <CButtonStateTitle>hover</CButtonStateTitle>
                            <div :style="getGhostTypeBackgroundColor(type)">
                                <template v-for="(color, index) in getFilterButtonColors(type)">
                                    <CButton :color="color" :type="type" class="mb-16 hover-test">Button</CButton>
                                </template>
                            </div>
                        </div>
                        <div class="flex-column mr-30">
                            <CButtonStateTitle>disabled</CButtonStateTitle>
                            <div :style="getGhostTypeBackgroundColor(type)">
                                <template v-for="(color, index) in getFilterButtonColors(type)">
                                    <CButton :color="color" :type="type" disabled class="mb-16">Button</CButton>
                                </template>
                            </div>
                        </div>
                        <div class="flex-column mr-30">
                            <CButtonStateTitle>loading</CButtonStateTitle>
                            <div :style="getGhostTypeBackgroundColor(type)">
                                <template v-for="(color, index) in getFilterButtonColors(type)">
                                    <CButton :color="color" :type="type" class="mb-16" loading style="color: transparent;">Button</CButton>
                                </template>
                            </div>
                        </div>
                    </div>
                  </template>
                </div>
            `,
            data() {
                return {
                    buttonColors,
                    buttonTypes,
                }
            },
            methods: {
                getFilterButtonColors(type) {
                    if (type==='ghost') {
                        return ghostTypeButtonColors;
                    }
                    return type === 'fill' ? this.buttonColors : this.filteredButtonColors
                },
                getGhostTypeBackgroundColor(type) {
                    return type === 'ghost' ? 'backgroundColor: #EE7E41' : ''
                },
            },
            computed: {
                // light-primary는 fill type에서만 쓰임
                filteredButtonColors() {
                    return Object.assign(buttonColors, {}).filter(item => item !== 'light-primary')
                },
            },
            components: { CButton, CTypography,
                'ButtonStateTitle': {
                    template: `<CTypography type="body2" color="gray400" class="mb-10" style="min-width: 70px"><slot/></CTypography>`,
                    components: {CTypography},
                },
            },
        }}
    </Story>
</Canvas>
