export const Default = (args, { argTypes }) => ({
    props: Object.keys(argTypes),
    components: { CCallout },
    data() {
        return {
            showCallout: true,
        }
    },
    template: `
        <CCallout
            v-if="showCallout"
            :type="type" :size="size"
            :full="full"
            :closable="closable"
            @closeCallout="showCallout = false"
        >
            {{ text }}
        </CCallout>`,
});
