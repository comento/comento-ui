export const Default = (args, { argTypes }) => ({
    props: Object.keys(argTypes),
    components: { Callout },
    data() {
        return {
            showCallout: true,
        }
    },
    template: `
        <Callout
            v-if="showCallout"
            :type="type" :size="size"
            :full="full"
            :closable="closable"
            @closeCallout="showCallout = false"
        >
            {{ text }}
        </Callout>`,
});
