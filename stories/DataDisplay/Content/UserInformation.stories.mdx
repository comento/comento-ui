    props: Object.keys(argTypes),
    components: { UserInformation, Content, Avatar, Icon, ChipGroup, Chip },
    data() {
        return {
            matchedItems: ['회사', '산업', '직무', '학교'],
        };
    },
    computed: {
        avatarSize() {
            const sizes = {
                simple: 'xsmall',
                normal: 'small',
                full: 'medium'
            };
            return sizes[this.type];
        }
    },
    template: `
        <UserInformation :type="type">

                        size="small"
