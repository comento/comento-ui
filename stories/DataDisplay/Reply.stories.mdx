import { ArgsTable, Meta, Story, Canvas } from "@storybook/addon-docs/blocks";
import Reply from "@/src/Components/DataDisplay/Reply";
import UserInformation from "@/src/Components/DataDisplay/Content/UserInformation";
import Avatar from "@/src/Components/DataDisplay/Avatar";
import Icon from "@/src/Elements/Core/Icon/Icon";

<Meta
    title="Data Display/Reply"
    component={ Reply }
    argTypes={{
        value: {
            description: "댓글 내용",
            defaultValue: "댓글 내용이 표시됩니다.",
            control: { type: "text" },
        },
        date: {
            description: "날짜",
            defaultValue: "21.09.02",
            control: { type: "text" },
        },
        type: {
            description: "댓글 타입",
            defaultValue: "default",
            control: {
                type: "select",
                options: [
                    "default",
                    "simple"
                ],
            }
        }
    }}
/>

export const Basic = (args, { argTypes }) => ({
    props: Object.keys(argTypes),
    components: { Reply, UserInformation, Avatar, Icon },
    data() {
        return {
            name: '닉네임',
            company: '코멘토',
        };
    },
    computed: {
        avatarSize() {
            const sizes = {
                default: 'small',
                simple: 'xsmall',
            }
            return sizes[this.type];
        },
        userInformationType() {
            if (this.type === 'default') {
                return 'normal';
            }
            return this.type;
        }
    },
    template: `
        <Reply :value="value" :date="date">
            <template v-slot:user-information>
                <UserInformation :type="userInformationType">
                    <template v-slot:avatar>
                        <Avatar type="text" :text="name.substr(0, 1)" :size="avatarSize"/>
                    </template>
                    <template v-slot:name>
                        {{ name }}
                    </template>
                    <template v-slot:icon>
                        <Icon name="IconCheckSmallFill" color="primary"/>
                    </template>
                    <template v-slot:company>
                        {{ company }}
                    </template>
                </UserInformation>
            </template>
        </Reply>`,
});
